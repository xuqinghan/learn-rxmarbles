# plot marble diagram

plot marble diagram for reactiveX by rxmarbles  [https://bitbucket.org/achary/rx-marbles/src/master/]

# install rxmarbles

pip install rxmarbles

# syntaxs


timeline:
- starting time point: .
- start: + 
- step: -
- end:
    - simple: >
    - complete: |
    - error: #

item:
    - single character: (a-z, A-Z, 0-9)
    - multiple character: ()

2-nd order timelines: 
source name: {}

Grouped items:
{1,2}

Operators:
operator myTransformation:  +-A-B-C-|

commet: //
